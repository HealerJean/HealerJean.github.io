# 一、数据库预热

## 1、动态数据源预热

```java
/**
 * 数据源预热
 *
 * @author zhangyujin
 * @date 2026/1/15
 */
@Service
public class DataSourceWarmUp implements CommandLineRunner {


    /**
     * dataSourceStartWarmUp
     */
    @Resource
    private DataSourceStartWarmUp dataSourceStartWarmUp;

    /**
     * 数据源预热
     */
    @Override
    public void run(String... args) throws Exception {
        dataSourceStartWarmUp.warmUp();
    }

}




import com.baomidou.dynamic.datasource.DynamicRoutingDataSource;
import com.jd.merchant.business.platform.core.common.DbConstants;
import lombok.extern.slf4j.Slf4j;
import org.springframework.retry.annotation.Backoff;
import org.springframework.retry.annotation.Recover;
import org.springframework.retry.annotation.Retryable;
import org.springframework.stereotype.Service;

import javax.annotation.Resource;
import javax.sql.DataSource;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

/**
 * DataSourceStartWarmUp
 *
 * @author zhangyujin
 * @date 2026/1/15
 */
@Service
@Slf4j
public class DataSourceStartWarmUp {
    /**
     * dynamicRoutingDataSource
     */
    @Resource
    private DynamicRoutingDataSource dynamicRoutingDataSource;

    /**
     * 数据源预热
     */
    @Retryable(value = Exception.class, maxAttempts = 4, backoff = @Backoff(delay = 20000, multiplier = 1.5))
    public void warmUp() {
        DataSource targetDs = dynamicRoutingDataSource.getDataSource(DbConstants.DataSourceName);
        if (targetDs == null) {
            throw new IllegalStateException("数据源 [" + DbConstants.DataSourceName + "] 未找到");
        }

        try (Connection conn = targetDs.getConnection();
             PreparedStatement ps = conn.prepareStatement("SELECT 1");
             ResultSet rs = ps.executeQuery()) {
            if (rs.next()) {
                log.info("数据源 [{}] 预热成功", DbConstants.DataSourceName);
            }
        } catch (Exception e) {
            log.warn("数据源 [{}] 预热失败，原因:{}", DbConstants.DataSourceName, e.getMessage());
            throw new RuntimeException("数据源预热失败，触发重试", e);
        }
    }

    /**
     * 数据源预热后失败了
     */
    @Recover
    public void recoverWarmUp(Exception ex) {
        log.error("所有重试均失败，数据源预热彻底失败", ex);
        throw new RuntimeException("数据源预热最终失败，应用无法安全启动", ex);
    }
}


```



## 2、普通预热

```java

import lombok.extern.slf4j.Slf4j;
import org.springframework.retry.annotation.Backoff;
import org.springframework.retry.annotation.Recover;
import org.springframework.retry.annotation.Retryable;
import org.springframework.stereotype.Service;

import javax.annotation.Resource;
import javax.sql.DataSource;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

/**
 * CDS数据源预热
 * <p>
 * 所属业务模块：基础设施-预热
 * 核心职责：对CDS主从数据源进行连接预热，确保应用启动后连接池已初始化，
 *          避免首次业务请求时因连接建立导致的延迟超时。
 * </p>
 *
 * @author zhangyujin
 * @date 2026/6/9
 */
@Service
@Slf4j
public class DbWarmUp {

    /**
     * 动态路由数据源（包含master和slave_router_all）
     */
    @Resource
    private DataSource dataSource;

    /**
     * 数据源预热
     * <p>
     * 执行流程：
     * <ul>
     *     <li>1. 从动态路由数据源获取连接</li>
     *     <li>2. 执行 SELECT 1 验证连接可用性</li>
     *     <li>3. 失败时自动重试（最多4次，间隔20s，1.5倍退避）</li>
     * </ul>
     * </p>
     */
    @Retryable(value = Exception.class, maxAttempts = 4, backoff = @Backoff(delay = 20000, multiplier = 1.5))
    public void warmUp() {
        // 1. 执行简单查询验证数据源连接
        //    1) 获取连接并执行 SELECT 1，触发连接池初始化
        try (Connection conn = dataSource.getConnection();
             PreparedStatement ps = conn.prepareStatement("SELECT 1");
             ResultSet rs = ps.executeQuery()) {
            if (rs.next()) {
                log.info("[DbWarmUp#warmUp] CDS数据源预热成功");
            }
        } catch (Exception e) {
            log.warn("[DbWarmUp#warmUp] CDS数据源预热失败，原因:{}", e.getMessage());
            throw new RuntimeException("CDS数据源预热失败，触发重试", e);
        }
    }

    /**
     * 数据源预热重试全部失败后的恢复处理
     *
     * @param ex 最后一次重试的异常信息
     */
    @Recover
    public void recoverWarmUp(Exception ex) {
        log.error("[DbWarmUp#recoverWarmUp] 所有重试均失败，CDS数据源预热彻底失败", ex);
        throw new RuntimeException("CDS数据源预热最终失败，应用无法安全启动", ex);
    }
}
```



# 二、缓存预热

```java
package com.jd.baoxian.merchant.route.web.configuration.warmup;

import com.wangyin.rediscluster.client.CacheClusterClient;
import lombok.extern.slf4j.Slf4j;
import org.springframework.retry.annotation.Backoff;
import org.springframework.retry.annotation.Recover;
import org.springframework.retry.annotation.Retryable;
import org.springframework.stereotype.Service;

import javax.annotation.Resource;

/**
 * R2M Redis集群预热
 * <p>
 * 所属业务模块：基础设施-预热
 * 核心职责：对R2M Redis集群客户端进行连接预热，确保应用启动后Redis连接池已初始化，
 *          避免首次业务请求时因Redis连接建立导致的延迟超时。
 * </p>
 *
 * @author zhangyujin
 * @date 2026/6/9
 */
@Service
@Slf4j
public class RedisWarmUp {

    /**
     * R2M Redis集群客户端
     */
    @Resource
    private RedisService redisService;

    /**
     * Redis预热键（使用专用前缀，不影响业务数据）
     */
    private static final String WARM_UP_KEY = "warmup:r2m:ping";

    /**
     * R2M Redis集群预热
     * <p>
     * 执行流程：
     * <ul>
     *     <li>1. 通过 setex 写入一个短生命周期的预热键验证连接可用性</li>
     *     <li>2. 失败时自动重试（最多4次，间隔20s，1.5倍退避）</li>
     * </ul>
     * </p>
     */
    @Retryable(value = Exception.class, maxAttempts = 4, backoff = @Backoff(delay = 20000, multiplier = 1.5))
    public void warmUp() {
        // 1. 执行简单写入操作验证Redis连接
        //    1) 写入短生命周期的键值对，触发连接池初始化
        try {
            String result = cacheClusterClient.setex(WARM_UP_KEY, 60, "1");
            log.info("[RedisWarmUp#warmUp] R2M Redis集群预热成功, result:{}", result);
        } catch (Exception e) {
            log.warn("[RedisWarmUp#warmUp] R2M Redis集群预热失败，原因:{}", e.getMessage());
            throw new RuntimeException("R2M Redis集群预热失败，触发重试", e);
        }
    }

    /**
     * R2M Redis预热重试全部失败后的恢复处理
     *
     * @param ex 最后一次重试的异常信息
     */
    @Recover
    public void recoverWarmUp(Exception ex) {
        log.error("[RedisWarmUp#recoverWarmUp] 所有重试均失败，R2M Redis集群预热彻底失败", ex);
        throw new RuntimeException("R2M Redis集群预热最终失败，应用无法安全启动", ex);
    }
}
```
